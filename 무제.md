public class Calculator {
	String operator = "+";

	Calculator(String operator){
		this.operator = operator;
	}
	
	double calculate(double x, double y) { 
		if (operator == "+") {
			return x + y;
		} else if (operator == "-") {
			return x - y;
		} else if (operator == "*") {
			return x * y;
		} else {
			return x / y;
		}
	}
	
	void setOperator(String operator) {
		this.operator = operator;
	}
}
